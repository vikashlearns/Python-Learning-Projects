#simple Calculator

num1=float(input("Enter first number: "))

addition=num1+num2

print("Addition:", addition)
print("Subtraction:", subtraction)

print("Division:", division)
print("Floor Division:", floor_division)

print("Exponentiation:", exponentiation)
